async function fetchPrice(appid){ const cc = currentCC(); const r = await fetch(`${API}/steam/price?appid=${appid}&cc=${encodeURIComponent(cc)}`, { cache: "no-store" }); const j = await r.json(); if(!r.ok || !j.ok) return null; // Sadece oyun istiyoruz: // Steam types: game, dlc, demo, soundtrack, application, video, series, episode... if(j.type !== "game") return { drop: true }; // Free ve fiyat yoklar zaten steam:null dönüyor if(!j.steam || !j.steam.final || j.steam.final <= 0) return { drop: true }; return { name: j.name || null, steam: j.steam }; }
